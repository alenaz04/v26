-- Создание базы данных
CREATE DATABASE demkaDB;
GO

USE demkaDB;
GO

-- 1. Таблица Customer (Клиенты)
CREATE TABLE Customer (
    id INT IDENTITY(1,1) PRIMARY KEY,
    name NVARCHAR(100) NOT NULL,
    inn BIGINT NOT NULL,
    address NVARCHAR(200),
    phone NVARCHAR(20),
    salesman BIT NOT NULL DEFAULT 0,
    buyer BIT NOT NULL DEFAULT 0
);
GO

-- 2. Таблица Role (Роли)
CREATE TABLE Role (
    id INT IDENTITY(1,1) PRIMARY KEY,
    title NVARCHAR(50) NOT NULL
);
GO

-- 3. Таблица User (Пользователи)
CREATE TABLE [User] (
    id INT IDENTITY(1,1) PRIMARY KEY,
    login NVARCHAR(50) NOT NULL UNIQUE,
    password NVARCHAR(100) NOT NULL,
    block BIT NOT NULL DEFAULT 0,
    role_id INT NOT NULL,
    FOREIGN KEY (role_id) REFERENCES Role(id)
);
GO

-- 4. Таблица Product (Продукты)
CREATE TABLE Product (
    id INT IDENTITY(1,1) PRIMARY KEY,
    name NVARCHAR(100) NOT NULL,
    unit NVARCHAR(20) NOT NULL,
    price DECIMAL(18,2) NOT NULL DEFAULT 0
);
GO

-- 5. Таблица Material (Материалы)
CREATE TABLE Material (
    id INT IDENTITY(1,1) PRIMARY KEY,
    name NVARCHAR(100) NOT NULL,
    unit NVARCHAR(20) NOT NULL,
    price DECIMAL(18,2) NOT NULL DEFAULT 0
);
GO

-- 6. Таблица Order (Заказы)
CREATE TABLE [Order] (
    id INT IDENTITY(1,1) PRIMARY KEY,
    date DATETIME NOT NULL DEFAULT GETDATE(),
    unit NVARCHAR(50),
    customer_id INT NOT NULL,
    FOREIGN KEY (customer_id) REFERENCES Customer(id)
);
GO

-- 7. Таблица OrderItem (Позиции заказа)
CREATE TABLE OrderItem (
    id INT IDENTITY(1,1) PRIMARY KEY,
    product_id INT NOT NULL,
    order_id INT NOT NULL,
    quantity DECIMAL(18,2) NOT NULL,
    sale DECIMAL(5,2) NOT NULL DEFAULT 0,
    FOREIGN KEY (product_id) REFERENCES Product(id),
    FOREIGN KEY (order_id) REFERENCES [Order](id)
);
GO

-- 8. Таблица Specification (Спецификации)
CREATE TABLE Specification (
    id INT IDENTITY(1,1) PRIMARY KEY,
    name NVARCHAR(100) NOT NULL,
    product_id INT NOT NULL,
    FOREIGN KEY (product_id) REFERENCES Product(id)
);
GO

-- 9. Таблица SpecificationItem (Позиции спецификации)
CREATE TABLE SpecificationItem (
    id INT IDENTITY(1,1) PRIMARY KEY,
    material_id INT NOT NULL,
    quantity DECIMAL(18,2) NOT NULL,
    specification_id INT NOT NULL,
    FOREIGN KEY (material_id) REFERENCES Material(id),
    FOREIGN KEY (specification_id) REFERENCES Specification(id)
);
GO

-- 10. Таблица Production (Производство)
CREATE TABLE Production (
    id INT IDENTITY(1,1) PRIMARY KEY,
    quantity DECIMAL(18,2) NOT NULL,
    date DATETIME NOT NULL DEFAULT GETDATE(),
    product_id INT NOT NULL,
    FOREIGN KEY (product_id) REFERENCES Product(id)
);
GO

-- 11. Таблица ProductionMaterial (Использованные материалы в производстве)
CREATE TABLE ProductionMaterial (
    id INT IDENTITY(1,1) PRIMARY KEY,
    production_id INT NOT NULL,
    material_id INT NOT NULL,
    quantity DECIMAL(18,2) NOT NULL,
    FOREIGN KEY (production_id) REFERENCES Production(id),
    FOREIGN KEY (material_id) REFERENCES Material(id)
);
GO

-- 12. Таблица sysdiagrams (для диаграмм базы данных)
CREATE TABLE sysdiagrams (
    name NVARCHAR(128) NOT NULL,
    principal_id INT NOT NULL,
    diagram_id INT PRIMARY KEY IDENTITY(1,1),
    version INT,
    definition VARBINARY(MAX)
);
GO

-- Создание индексов для оптимизации
CREATE INDEX IX_Order_customer_id ON [Order](customer_id);
CREATE INDEX IX_OrderItem_order_id ON OrderItem(order_id);
CREATE INDEX IX_OrderItem_product_id ON OrderItem(product_id);
CREATE INDEX IX_Production_product_id ON Production(product_id);
CREATE INDEX IX_ProductionMaterial_production_id ON ProductionMaterial(production_id);
CREATE INDEX IX_ProductionMaterial_material_id ON ProductionMaterial(material_id);
CREATE INDEX IX_Specification_product_id ON Specification(product_id);
CREATE INDEX IX_SpecificationItem_specification_id ON SpecificationItem(specification_id);
CREATE INDEX IX_SpecificationItem_material_id ON SpecificationItem(material_id);
CREATE INDEX IX_User_role_id ON [User](role_id);
GO
